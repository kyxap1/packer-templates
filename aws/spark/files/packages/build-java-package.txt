
# create the build folder and change into it
mkdir -p packages/java
cd packages/java

# get the java archive
curl -L -b "oraclelicense=a" http://download.oracle.com/otn-pub/java/jdk/8u92-b14/server-jre-8u92-linux-x64.tar.gz

# install the required tools to build the deb from the archive
sudo apt-get install java-package

# build the deb file
make-jpkg server-jre-8u92-linux-x64.tar.gz
